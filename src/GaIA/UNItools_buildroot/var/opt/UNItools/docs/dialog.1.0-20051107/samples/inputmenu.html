0 #! /bin/bash
    1 # $Id: inputmenu,v 1.6 2004/09/19 17:59:14 tom Exp $
    2 # 2002 - written by Tobias Rittweiler <tobrit@freebits.de>
    3 : ${DIALOG=dialog}
    4 
    5 user="$USER"    ; uid="$UID"   ;
    6 gid="$GROUPS"  ; home="$HOME"  ;
    7 
    8 while [ ${returncode:-99} -ne 1 -a ${returncode:-99} -ne 250 ]; do
    9        exec 3>&1
   10        value=$("$DIALOG" --clear --ok-label "Create" \
   11                           --backtitle "An Example for the use of --inputmenu:" \
   12                           --inputmenu "Originally I designed --inputmenu for a \
   13 configuration purpose. Here is a possible piece of a configuration program. \
   14 " 20 50 10 \
   15 "Username:" "$user" \
   16 "UID:"      "$uid" \
   17 "GID:"      "$gid" \
   18 "HOME:"     "$home" \
   19 2>&1 1>&3 )
   20        returncode=$?
   21        exec 3>&-
   22        case $returncode in
   23        1)
   24                "$DIALOG"  --clear --backtitle "An Example for the use of --inputmenu:" \
   25                                --yesno "Really quit?" 10 30
   26                case $? in
   27                        0) break;;
   28                        1) returncode=99;;
   29                esac
   30                ;;
   31        0)
   32                "$DIALOG" --clear --backtitle "An Example for the use of --inputmenu:"  \
   33 			--msgbox "useradd \n\
   34 				-d $home \n\
   35 				-u $uid \n\
   36 				-g $gid \n\
   37 				$user" 10 40
   38                ;;
   39        3)
   40                value="${value:8:${#value}}"
   41                tag="${value%:*}"
   42                item="${value#*: }"
   43 
   44                case "$tag" in
   45                        Username) user="$item";;
   46                        UID) uid="$item";;
   47                        GID) gid="$item";;
   48                        HOME) home="$item";;
   49                esac
   50                ;;
   51 
   52        255)
   53                 echo "ESC pressed."
   54                 break
   55                 ;;
   56 
   57        esac
   58 done
