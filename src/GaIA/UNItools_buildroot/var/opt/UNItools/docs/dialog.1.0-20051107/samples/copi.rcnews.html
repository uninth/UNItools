0 #! /bin/sh
    1 ##  $Revision: 1.1 $
    2 ##  News boot script.
    3 
    4 ##  =()<. @<_PATH_SHELLVARS>@>()=
    5 . /usr/lib/news/innshellvars
    6 
    7 ##  Pick ${INND} or ${INNDSTART}
    8 WHAT=${INNDSTART}
    9 ##  Set to true or false
   10 DOINNWATCH=false
   11 MAIL="${MAILCMD} -s 'Boot-time Usenet warning on `hostname`' ${NEWSMASTER}"
   12 
   13 ##  RFLAG is set below; set FLAGS as appropriate.
   14 RFLAG=""
   15 FLAGS="-i0 -c0"
   16 
   17 ##  Clean shutdown?
   18 if [ -f ${SERVERPID} ] ; then
   19     ( echo 'INND:  PID file exists -- unclean shutdown!' >/dev/console )
   20     RFLAG="-r"
   21 fi
   22 
   23 if [ ! -f ${NEWSLIB}/.news.daily ] ; then
   24     echo 'No .news.daily file; need to run news.daily?' | eval ${MAIL}
   25 else
   26     case `find ${NEWSLIB} -name .news.daily -mtime +1 -print 2>/dev/null` in
   27     "")
   28 	;;
   29     *)
   30 	echo 'Old .news.daily file; need to run news.daily?' | eval ${MAIL}
   31 	;;
   32     esac
   33 fi
   34 
   35 ##  Active file recovery.
   36 if [ ! -s ${ACTIVE} ] ; then
   37     if [ -s ${NEWACTIVE} ] ; then
   38 	mv ${NEWACTIVE} ${ACTIVE}
   39     else
   40 	if [ -s ${OLDACTIVE} ] ; then
   41 	    cp ${OLDACTIVE} ${ACTIVE}
   42 	else
   43 	    ( echo 'INND:   No active file!' >/dev/console )
   44 	    exit 1
   45 	fi
   46     fi
   47     RFLAG="-r"
   48     # You might want to rebuild the DBZ database, too:
   49     #echo "cd ${NEWSLIB} \
   50     #		&& makehistory -r \
   51     #		&& mv history.n.dir history.dir \
   52     #		&& mv history.n.pag history.pag" | su ${NEWSUSER}
   53 fi
   54 
   55 ##  Remove temporary batchfiles and lock files.
   56 ( cd ${BATCH} && rm -f bch* )
   57 ( cd ${LOCKS} && rm -f LOCK* )
   58 ( cd ${TEMPSOCKDIR} && rm -f ${TEMPSOCK} )
   59 rm -f ${NEWSCONTROL} ${NNTPCONNECT} ${SERVERPID}
   60 
   61 ##  Start the show.
   62 ( echo 'Starting innd.' >/dev/console )
   63 eval ${WHAT} ${RFLAG} ${FLAGS}
   64 
   65 # Gee, looks like lisp, doesn't it?
   66 ${DOINNWATCH} && {
   67     echo "( ( sleep 60 ; ${INNWATCH} ) & )" | su ${NEWSUSER}
   68 }
   69 
   70 RMFILE=${MOST_LOGS}/expire.rm
   71 if [ -s ${MOST_LOGS}/expire.rm ] ; then
   72     ( echo "Removing articles from pre-downtime expire run." >/dev/console )
   73     (
   74 	echo 'System shut down during expire.  Unlinking articles listed in'
   75 	echo ${RMFILE}
   76     ) | eval ${MAIL}
   77     echo "${NEWSBIN}/expirerm ${RMFILE}" | su ${NEWSUSER} &
   78 fi
