0 #!/bin/sh
    1 # $Id: menubox6,v 1.3 2004/12/19 23:18:53 tom Exp $
    2 # example showing the --colors option
    3 : ${DIALOG=dialog}
    4 
    5 tempfile=`tempfile 2>/dev/null` || tempfile=/tmp/test$$
    6 trap "rm -f $tempfile" 0 1 2 5 15
    7 
    8 $DIALOG --clear --item-help --title "MENU BOX" --colors \
    9         --menu "Hi, this is a \Zumenu\Zn box. You can use this to \n\
   10 present a list of choices for the user to \n\
   11 choose. If there are more items than can fit \n\
   12 on the screen, the menu will be scrolled. \n\
   13 You can use the \Z4UP/DOWN\Zn arrow keys, the first \n\
   14 letter of the choice as a hot key, or the \n\
   15 \Z2number\Zn keys 1-9 to choose an option.\n\
   16 Try it now!\n\n\
   17           Choose the OS you like:" 20 51 4 \
   18         "Linux"  "The Great Unix Clone for \Zu386/486"    "Why use \ZbLinux?" \
   19         "NetBSD" "Another free Unix Clone for \Zu386/486" "Or \ZbNetBSD?" \
   20         "OS/2"   "IBM OS/2"                            "aka \"Warp\"" \
   21         "WIN NT" "Microsoft Windows NT"                "hmm" \
   22         "PCDOS"  "IBM PC \ZrDOS"                          "clone of a clone" \
   23         "MSDOS"  "Microsoft \ZrDOS"                       "\ZbDOS\Zn: \Zb\Z3D\Znisk \Zb\Z3O\Znperating \Zb\Z3S\Znystem, originally for an IBM contract, hence using the same jargon" 2> $tempfile
   24 
   25 retval=$?
   26 
   27 choice=`cat $tempfile`
   28 
   29 case $retval in
   30   0)
   31     echo "OK '$choice' chosen.";;
   32   1)
   33     echo "Cancel pressed.";;
   34   2)
   35     echo "HELP '$choice' chosen.";;
   36   255)
   37     echo "ESC pressed.";;
   38   *)
   39     echo "Unexpected code $retval";;
   40 esac
