0 #!/bin/sh
    1 # $Id: tailbox,v 1.3 2003/08/15 19:40:37 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 MYNAME="UNItools system monitoring tools"
    5 VERSION="1.0"
    6 
    7 PROG=itop
    8 EXEC=/var/opt/UNItools/bin/$PROG
    9 USAGE="`$PROG -h`"
   10 WHATIS="itop - top-like interrupt load monitor"
   11 MAN="/var/opt/UNItools/docs/SPM/itop.0.1.1/itop.txt"
   12 
   13 USAGE_FILE=/tmp/tmpfile
   14 $USAGE > $USAGE_FILE 2>&1
   15 
   16 # select / menubox1
   17 $DIALOG --clear --backtitle "$MYNAME - $VERSION" \
   18 	--title "Main UNItools menu" \
   19         --menu "Use arrow keys, mouse, number keys 1-9 or \
   20 first letter to choose an option.\n" 20 70 10 \
   21 "$PROG"  "$WHATIS" \
   22 "MSDOS"  "Microsoft DOS" 2> /dev/null
   23 
   24 # tool/usage/man
   25 $DIALOG --clear --backtitle "$MYNAME - $VERSION" \
   26 	--title "$PROG / Usage / Manual" \
   27 	--default-item "$USAGE" \
   28         --menu "Use arrow keys, mouse, number keys 1-9 or \
   29 first letter to choose an option.\n" 20 70 10 \
   30 "Run $PROG ... " "" \
   31 "Command Usage ... " "" \
   32 "Command manual ... " "" #2> /dev/null
   33 
   34 # man page
   35 $DIALOG --backtitle "$MYNAME - $VERSION"	\
   36 	--title "$WHATIS"	\
   37         --textbox  $MAN 24 80
   38 
   39 case $? in
   40   0)
   41     echo "OK";;
   42   255)
   43     echo "ESC pressed.";;
   44 esac
   45
