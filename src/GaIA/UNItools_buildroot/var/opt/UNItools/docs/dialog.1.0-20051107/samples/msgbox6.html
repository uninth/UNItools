0 #!/bin/sh
    1 # $Id: msgbox6,v 1.1 2004/11/18 00:57:40 tom Exp $
    2 # this differs from msgbox3 by making a window small enough to force scrolling.
    3 : ${DIALOG=dialog}
    4 
    5 width=35
    6 while test $width != 61
    7 do
    8 $DIALOG --title "MESSAGE BOX (width $width)" --clear \
    9         --msgbox "\
   10 	a b c d e f g h j i j k l m n o p q r s t u v w x y z
   11 	A B C D E F G H J I J K L M N O P Q R S T U V W X Y Z
   12 " 10 $width
   13 test $? = 255 && break
   14 width=`expr $width + 1`
   15 done
