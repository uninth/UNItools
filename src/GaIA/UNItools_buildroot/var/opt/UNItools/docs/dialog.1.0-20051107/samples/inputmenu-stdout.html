0 #! /bin/bash
    1 # $Id: inputmenu-stdout,v 1.4 2005/02/06 22:51:23 tom Exp $
    2 # 2002 - written by Tobias Rittweiler <tobrit@freebits.de>
    3 : ${DIALOG=dialog}
    4 
    5 user="$USER"    ; uid="$UID"   ;
    6 gid="$GROUPS"  ; home="$HOME"  ;
    7 
    8 while [ ${returncode:-99} -ne 1 -a ${returncode:-99} -ne 250 ]; do
    9        value="$("$DIALOG" --stdout --clear --ok-label "Create" \
   10                           --backtitle "An Example for the use of --inputmenu:" \
   11                           --inputmenu "Originally I designed --inputmenu for a \
   12 configuration purpose. Here is a possible piece of a configuration program.
   13 " 20 50 10 "Username:" "$user" "UID:" "$uid" "GID:" "$gid" "HOME:" "$home")"
   14        returncode=$?
   15        case $returncode in
   16            1)
   17                "$DIALOG"  --clear --backtitle "An Example for the use of --inputmenu:" \
   18                                --yesno "Really quit?" 10 30
   19                case $? in
   20                        0) break;;
   21                        1) returncode=99;;
   22                esac
   23                ;;
   24            0)
   25                "$DIALOG" --clear --backtitle "An Example for the use of --inputmenu:"  \
   26 			--msgbox "useradd \n\
   27 				-d $home \n\
   28 				-u $uid \n\
   29 				-g $gid \n\
   30 				$user" 10 40
   31                ;;
   32            3)
   33                value="${value:8:${#value}}"
   34                tag="${value%:*}"
   35                item="${value#*: }"
   36 
   37                case "$tag" in
   38                        Username) user="$item";;
   39                        UID) uid="$item";;
   40                        GID) gid="$item";;
   41                        HOME) home="$item";;
   42                esac
   43 	       ;;
   44 
   45        255)
   46                 echo "ESC pressed."
   47                 break
   48                 ;;
   49 
   50        esac
   51 done
