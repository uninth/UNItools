0 #!/bin/sh
    1 # $Id: fselect1-stdout,v 1.1 2003/08/15 19:40:37 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 FILE=$HOME
    5 for n in .cshrc .profile .bashrc
    6 do
    7 	if test -f $HOME/$n ; then
    8 		FILE=$HOME/$n
    9 		break
   10 	fi
   11 done
   12 
   13 FILE=`$DIALOG --stdout --title "Please choose a file" --fselect $FILE 14 48`
   14 
   15 case $? in
   16 	0)
   17 		echo "\"$FILE\" chosen";;
   18 	1)
   19 		echo "Cancel pressed.";;
   20 	255)
   21 		echo "Box closed.";;
   22 esac
