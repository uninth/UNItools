0 #!/bin/sh
    1 # $Id: inputbox,v 1.6 2003/08/15 19:40:37 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 tempfile=`tempfile 2>/dev/null` || tempfile=/tmp/test$$
    5 trap "rm -f $tempfile" 0 1 2 5 15
    6 
    7 $DIALOG --title "INPUT BOX" --clear \
    8         --inputbox "Hi, this is an input dialog box. You can use \n
    9 this to ask questions that require the user \n
   10 to input a string as the answer. You can \n
   11 input strings of length longer than the \n
   12 width of the input box, in that case, the \n
   13 input field will be automatically scrolled. \n
   14 You can use BACKSPACE to correct errors. \n\n
   15 Try entering your name below:" 16 51 2> $tempfile
   16 
   17 retval=$?
   18 
   19 case $retval in
   20   0)
   21     echo "Input string is `cat $tempfile`";;
   22   1)
   23     echo "Cancel pressed.";;
   24   255)
   25     if test -s $tempfile ; then
   26       cat $tempfile
   27     else
   28       echo "ESC pressed."
   29     fi
   30     ;;
   31 esac
