0 #! /bin/sh
    1 # $Id: radiolist3,v 1.4 2004/12/20 00:14:24 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 tempfile=`tempfile 2>/dev/null` || tempfile=/tmp/test$$
    5 trap "rm -f $tempfile" 0 1 2 5 15
    6 
    7 $DIALOG --item-help --ok-label Okay \
    8 	--cancel-label 'Give Up' \
    9 	--backtitle "No Such Organization" \
   10 	--title "RADIOLIST BOX" --clear \
   11         --radiolist "Hi, this is a radiolist box. You can use this to \n\
   12 present a list of choices which can be turned on or \n\
   13 off. If there are more items than can fit on the \n\
   14 screen, the list will be scrolled. You can use the \n\
   15 UP/DOWN arrow keys, the first letter of the choice as a \n\
   16 hot key, or the number keys 1-9 to choose an option. \n\
   17 Press SPACE to toggle an option on/off. \n\n\
   18   Which of the following are fruits?" 20 61 5 \
   19         "Apple"  "It's an apple." off "Hint: this grows in a tree" \
   20         "Dog"    "No, that's not my dog." ON "Hint: this likes trees" \
   21         "Orange" "Yeah, that's juicy." off "Hint: this is green when picked" \
   22         "Chicken"    "Normally not a pet." off "Hint: not often in trees" \
   23         "Cat"    "No, never put a dog and a cat together!" off "Hint: may be found in trees" \
   24         "Fish"   "Cats like fish." off "Hint: usually not close to cats" \
   25         "Lemon"  "You know how it tastes." off "Hint: like an orange" 2> $tempfile
   26 
   27 retval=$?
   28 
   29 choice=`cat $tempfile`
   30 case $retval in
   31   0)
   32     echo "OK '$choice' chosen.";;
   33   1)
   34     echo "Cancel pressed.";;
   35   2)
   36     echo "HELP '$choice' chosen.";;
   37   255)
   38     echo "ESC pressed.";;
   39   *)
   40     echo "Unexpected code $retval";;
   41 esac
