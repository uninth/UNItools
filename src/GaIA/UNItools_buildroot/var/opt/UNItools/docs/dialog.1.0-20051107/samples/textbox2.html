0 #!/bin/sh
    1 # $Id: textbox2,v 1.2 2003/08/15 19:40:37 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 tempfile=`tempfile 2>/dev/null` || tempfile=/tmp/test$$
    5 trap "rm -f $tempfile" 0 1 2 5 15
    6 
    7 cat << EOF > $tempfile
    8 Hi, this is a text dialog box. It can be used to display text from a file.
    9 The file should not contain any 'tab' characters, so you should 'expand'
   10 the file first if it contains 'tab' characters.
   11 
   12 It's like a simple text file viewer, with these keys implemented:
   13 
   14 PGDN/SPACE     - Move down one page
   15 PGUP/'b'       - Move up one page
   16 ENTER/DOWN/'j' - Move down one line
   17 UP/'k'         - Move up one line
   18 LEFT/'h'       - Scroll left
   19 RIGHT/'l'      - Scroll right
   20 '0'            - Move to beginning of line
   21 HOME/'g'       - Move to beginning of file
   22 END/'G'        - Move to end of file
   23 '/'            - Forward search
   24 '?'            - Backward search
   25 'n'            - Repeat last search (forward)
   26 'N'            - Repeat last search (backward)
   27 
   28 
   29 The following is a sample text file:
   30 
   31 
   32 EOF
   33 
   34 TEXT=/usr/share/common-licenses/GPL
   35 test -f $TEXT || TEXT=../COPYING
   36 
   37 cat $TEXT | expand >> $tempfile
   38 
   39 $DIALOG --clear --title "TEXT BOX" --textbox "$tempfile" 0 0
   40 
   41 case $? in
   42   0)
   43     echo "OK";;
   44   255)
   45     echo "ESC pressed.";;
   46 esac
