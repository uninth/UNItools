0 #!/bin/sh
    1 # $Id: msgbox5,v 1.1 2004/06/05 17:19:43 tom Exp $
    2 # this differs from msgbox3 by making a window small enough to force scrolling.
    3 : ${DIALOG=dialog}
    4 
    5 width=35
    6 while test $width != 61
    7 do
    8 $DIALOG --title "MESSAGE BOX (width $width)" --clear --no-collapse \
    9         --msgbox "\
   10 	H   H EEEEE L     L      OOO
   11 	H   H E     L     L     O   O
   12 	HHHHH EEEEE L     L     O   O
   13 	H   H E     L     L     O   O
   14 	H   H EEEEE LLLLL LLLLL  OOO
   15 
   16 Hi, this is a simple message box.  You can use this to \
   17 display any message you like.  The box will remain until \
   18 you press the ENTER key." 10 $width
   19 test $? = 255 && break
   20 width=`expr $width + 1`
   21 done
