0 #!/bin/sh
    1 # $Id: msgbox4-utf8,v 1.6 2004/12/19 16:27:15 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 : ${DIALOG_OK=0}
    5 : ${DIALOG_CANCEL=1}
    6 : ${DIALOG_ESC=255}
    7 
    8 case none"$LANG$LC_ALL$LC_CTYPE" in
    9 *UTF-8*)
   10 	;;
   11 *)
   12 	echo "This script must be run in a UTF-8 locale"
   13 	exit 1
   14 	;;
   15 esac
   16 
   17 width=30
   18 while test $width != 61
   19 do
   20 $DIALOG --title "MESSAGE BOX (width $width)" --no-collapse \
   21         --msgbox "\
   22 This sample is written in UTF-8.
   23 There are several checking points:
   24 (1) whether the fullwidth characters are displayed well or not,
   25 (2) whether the width of characters are evaluated properly, and
   26 (3) whether the character at line-folding is lost or not.
   27 
   28 あいうえおかきくけこさしすせそたちつてとなにぬねの
   29 １２３４５６７８９０１２３４５６７８９０１２３４５
   30 ＡＢＣＤＥＦＧＨＩＪＫＬＭＮＯＰＱＲＳＴＵＶＷＸＹ
   31 
   32 Hi, this is a simple message box.  You can use this to \
   33 display any message you like.  The box will remain until \
   34 you press the ENTER key." 22 $width
   35 retval=$?
   36 
   37 case $retval in
   38   $DIALOG_CANCEL)
   39     echo "Cancel pressed.";exit;;
   40   $DIALOG_ESC)
   41     echo "ESC pressed.";exit;;
   42 esac
   43 
   44 width=`expr $width + 1`
   45 
   46 done
