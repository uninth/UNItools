    1 # Packets backup directory - You have to CREATE this directory
    2 BAK=/var/spool/ifmail/BAK
    3 
    4 # logfile of ifcico
    5 IFLOG=/var/log/ifmail/sysiflog
    6 
    7 # log archive
    8 IFLOGARC=/var/log/ifmail/TheLog
    9 
   10 # owner of "ifcico"
   11 IFCICO_OWNER=fnet
   12 
   13 # how often should i try to call NODE?
   14 MaxTry=30
   15 
   16 # delay between outgoing calls in seconds
   17 DELAY=10
   18 
   19 # where to log processing - file or tty/console
   20 INFO_TTY=/dev/console
   21 
   22 # Lets backup the log and use a new one
   23 if [ -e $IFLOG ]; then
   24 	echo Rearranging logs...
   25 	cat $IFLOG >> $IFLOGARC
   26 	cp /dev/null $IFLOG
   27 fi
   28 
   29 echo "`date \"+%b %d %T\"` ifpoll[$$]: starting" # >> $INFO_TTY
   30 
   31 # remember me, not to run as root..
   32 #
   33 if [ `whoami` != "$IFCICO_OWNER" ]; then
   34 	echo "*** run $0 as the owner of ifcico ***"
   35 	echo "`date \"+%b %d %T\"` ifpoll[$$]: wrong uid (rc 2)" # >> $INFO_TTY
   36 	exit 2
   37 fi
   38 
   39 # argv[1] is the optional node to call
   40 #
   41 if [ "$1" != "" ]; then
   42 	if [ "$1" = "-?" ] || [ "$1" = "-h" ]; then
   43 		echo "usage: ifpoll [<node>]"
   44 		exit 3
   45 	else
   46 		NODE=$1
   47 	fi
   48 fi
   49 
   50 # let's pack the fido stuff..
   51 #
   52 $FIDOPATH/ifpack
   53 
   54 # loop until ifcico could connect the node or MaxTry is encountered
   55 #
   56 i=1; errlv=1
   57 while let 'i <= MaxTry' && let 'errlv != 0'
   58 do
   59 	echo -n "`date \"+%b %d %T\"` ifpoll[$$]: $i. try ($NODE) " # >> $INFO_TTY
   60 	#
   61 	# start ifcico in master mode ..
   62 	#
   63 	$FIDOPATH/ifcico -r 1 $NODE
   64 	errlv=$?
   65 	if [ $errlv != "0" ]; then
   66 		echo "failed" # >> $INFO_TTY
   67 		if [ $i != $MaxTry ]; then
   68 			sleep $DELAY
   69 		fi
   70 		let i=i+1
   71 	else
   72 		echo "ok :)" # >> $INFO_TTY
   73 	fi
   74 done
   75 
   76 # if the poll was fine, unpacking..
   77 #
   78 if [ $errlv = "0" ]; then
   79 	echo "`date \"+%b %d %T\"` ifpoll[$$]: unpacking.. " # >> $INFO_TTY
   80     	cp -f /var/spool/ifmail/inb/*.* $BAK
   81 	$FIDOPATH/ifunpack
   82 	# add here some additional lines for processing tic files or
   83 	# incoming file-lists or simular..
   84 else
   85 	# write me a mail about the failed poll
   86 	tail --lines=20 $IFLOG | elm -s "ifpoll: failed" $IFCICO_SYSOP >/dev/null
   87 fi
   88 
   89 echo "`date \"+%b %d %T\"` ifpoll[$$]: finished (rc $errlv)" # >> $INFO_TTY
   90 
   91 # Collect some infos...
   92 echo ""
   93 echo "*** Outgoing File Transfer Stats: ***"
   94 grep sent $IFLOG
   95 echo "*** Incoming File Transfer Stats: ***"
   96 grep received $IFLOG
   97 echo ""
   98 
   99 echo "---->> New mail: <<-----"
  100 grep echomail $IFLOG
  101 echo "---->> Personal mail: <<-----"
  102 grep "mail from" $IFLOG
  103 echo ""
  104 
  105 # return the errorlevel of ifcico
  106 exit $errlv
