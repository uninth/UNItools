0 #! /bin/sh
    1 ##  $Revision: 1.1 $
    2 ##  SH script to send UUCP batches out.
    3 
    4 ##  =()<. @<_PATH_SHELLVARS>@>()=
    5 . ~news/innshellvars
    6 
    7 PROGNAME=`basename $0`
    8 LOCK=${LOCKS}/LOCK.${PROGNAME}
    9 LOG=${MOST_LOGS}/${PROGNAME}.log
   10 
   11 MAXJOBS=200
   12 UUXFLAGS="- -r -gd"
   13 
   14 ##  Start logging.
   15 test ! -f ${LOG} && touch ${LOG}
   16 chmod 0660 ${LOG}
   17 exec >>${LOG} 2>&1
   18 echo "${PROGNAME}: [$$] begin `date`"
   19 cd ${BATCH}
   20 
   21 ##  Anyone else there?
   22 trap 'rm -f ${LOCK} ; exit 1' 1 2 3 15
   23 shlock -p $$ -f ${LOCK} || {
   24     echo "${PROGNAME}: [$$] locked by [`cat ${LOCK}`]"
   25     exit 0
   26 }
   27 
   28 ##  Who are we sending to?
   29 if [ -n "$1" ] ; then
   30     LIST="$*"
   31 else
   32     LIST=`ls /usr/spool/news/out.going|egrep "f[0-9]*.n[0-9]*$"`
   33 fi
   34 
   35 case ${HAVE_UUSTAT} in
   36 DONT)
   37     TEMP=${TMPDIR}/uuq$$
   38     uuq -h |tr -d : >${TEMP}
   39     ;;
   40 esac
   41 
   42 ##  Do the work...
   43 
   44 for SITE in ${LIST}; do
   45 
   46     ##  See if any data is ready for host.
   47     BATCHFILE=${SITE}.uucp
   48     if [ -f ${SITE}.work ] ; then
   49 	cat ${SITE}.work >>${BATCHFILE}
   50 	rm -f ${SITE}.work
   51     fi
   52     mv ${SITE} ${SITE}.work
   53     ctlinnd -s -t30 flush ${SITE} || continue
   54     cat ${SITE}.work >>${BATCHFILE}
   55     rm -f ${SITE}.work
   56     if [ ! -s ${BATCHFILE} ] ; then
   57 	echo "${PROGNAME}: [$$] no articles for ${SITE}"
   58 	rm -f ${BATCHFILE}
   59 	continue
   60     fi
   61 
   62     ##  Get the compression flags.
   63     echo "${PROGNAME}: [$$] begin ${SITE}"
   64     case "${SITE}" in
   65     phreak)
   66 	COMPRESS="compress"
   67 	;;
   68     *)
   69 	COMPRESS="gzip"
   70 	;;
   71     esac
   72 
   73     time batcher \
