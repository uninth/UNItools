0 #!/bin/sh
    1 # $Id: fselect,v 1.5 2003/10/09 22:45:06 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 exec 3>&1
    5 FILE=`$DIALOG --title "Please choose a file" --fselect $HOME/ 14 48 2>&1 1>&3`
    6 code=$?
    7 exec 3>&-
    8 
    9 case $code in
   10 	0)
   11 		echo "\"$FILE\" chosen";;
   12 	1)
   13 		echo "Cancel pressed.";;
   14 	255)
   15 		echo "Box closed.";;
   16 esac
