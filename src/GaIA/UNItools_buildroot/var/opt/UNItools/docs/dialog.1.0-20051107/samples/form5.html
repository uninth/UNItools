0 #! /bin/sh
    1 # $Id: form5,v 1.2 2004/09/12 22:37:57 tom Exp $
    2 # form4 with --help-status
    3 : ${DIALOG=dialog}
    4 
    5 backtitle="An Example for the use of --form:"
    6 
    7 ids=`id|sed -e 's/([^)]*)//g'`
    8 uid=`echo "$ids" | sed -e 's/^uid=//' -e 's/ .*//'`
    9 gid=`echo "$ids" | sed -e 's/^.* gid=//' -e 's/ .*//'`
   10 
   11 user="$USER"
   12 home="$HOME"
   13 
   14 returncode=0
   15 while test $returncode != 1 && test $returncode != 250
   16 do
   17 exec 3>&1
   18 value=`$DIALOG --ok-label "Submit" \
   19 	  --help-status \
   20 	  --help-button \
   21 	  --item-help \
   22 	  --backtitle "$backtitle" \
   23 	  --form "Here is a possible piece of a configuration program." \
   24 20 50 0 \
   25 	"Username:" 1 1	"$user" 1 10 10 0 "Login name" \
   26 	"UID:"      2 1	"$uid"  2 10  8 0 "User ID" \
   27 	"GID:"      3 1	"$gid"  3 10  8 0 "Group ID" \
   28 	"HOME:"     4 1	"$home" 4 10 40 0 "User's home-directory" \
   29 2>&1 1>&3`
   30 returncode=$?
   31 exec 3>&-
   32 
   33 show=`echo "$value" |sed -e 's/^/	/'`
   34 
   35 	case $returncode in
   36 	1)
   37 		"$DIALOG" \
   38 		--clear \
   39 		--backtitle "$backtitle" \
   40 		--yesno "Really quit?" 10 30
   41 		case $? in
   42 		0)
   43 			break
   44 			;;
   45 		1)
   46 			returncode=99
   47 			;;
   48 		esac
   49 		;;
   50 	0)
   51 		"$DIALOG" \
   52 		--clear \
   53 		--backtitle "$backtitle" --no-collapse --cr-wrap \
   54 		--msgbox "Resulting data:\n\
   55 $show" 10 40
   56 		;;
   57 	2)
   58 		"$DIALOG" \
   59 		--clear \
   60 		--backtitle "$backtitle" --no-collapse --cr-wrap \
   61 		--msgbox "Help data:\n\
   62 $show" 10 40
   63 		;;
   64 	*)
   65 		echo "Return code was $returncode"
   66 		exit
   67 		;;
   68 	esac
   69 done
