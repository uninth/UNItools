0 #!/bin/sh
    1 # $Id: menubox3,v 1.3 2004/12/19 23:18:53 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 tempfile=`tempfile 2>/dev/null` || tempfile=/tmp/test$$
    5 trap "rm -f $tempfile" 0 1 2 5 15
    6 
    7 $DIALOG --clear --item-help --title "MENU BOX" \
    8         --menu "Hi, this is a menu box. You can use this to \n\
    9 present a list of choices for the user to \n\
   10 choose. If there are more items than can fit \n\
   11 on the screen, the menu will be scrolled. \n\
   12 You can use the UP/DOWN arrow keys, the first \n\
   13 letter of the choice as a hot key, or the \n\
   14 number keys 1-9 to choose an option.\n\
   15 Try it now!\n\n\
   16           Choose the OS you like:" 20 51 4 \
   17         "Linux"  "The Great Unix Clone for 386/486"    "Why use Linux?" \
   18         "NetBSD" "Another free Unix Clone for 386/486" "Or NetBSD?" \
   19         "OS/2"   "IBM OS/2"                            "aka \"Warp\"" \
   20         "WIN NT" "Microsoft Windows NT"                "hmm" \
   21         "PCDOS"  "IBM PC DOS"                          "clone of a clone" \
   22         "MSDOS"  "Microsoft DOS"                       "DOS: Disk Operating System, originally for an IBM contract, hence using the same jargon" 2> $tempfile
   23 
   24 retval=$?
   25 
   26 choice=`cat $tempfile`
   27 
   28 case $retval in
   29   0)
   30     echo "OK '$choice' chosen.";;
   31   1)
   32     echo "Cancel pressed.";;
   33   2)
   34     echo "HELP '$choice' chosen.";;
   35   255)
   36     echo "ESC pressed.";;
   37   *)
   38     echo "Unexpected code $retval";;
   39 esac
