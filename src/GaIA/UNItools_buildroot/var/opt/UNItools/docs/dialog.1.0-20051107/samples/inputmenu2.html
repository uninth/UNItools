0 #! /bin/sh
    1 # $Id: inputmenu2,v 1.4 2004/09/19 17:59:14 tom Exp $
    2 #
    3 # "inputmenu1" with defaultitem, help-button.
    4 : ${DIALOG=dialog}
    5 
    6 backtitle="An Example for the use of --inputmenu:"
    7 
    8 ids=`id|sed -e 's/([^)]*)//g'`
    9 uid=`echo "$ids" | sed -e 's/^uid=//' -e 's/ .*//'`
   10 gid=`echo "$ids" | sed -e 's/^.* gid=//' -e 's/ .*//'`
   11 
   12 user="$USER"
   13 home="$HOME"
   14 
   15 returncode=0
   16 defaultitem="Username:"
   17 while test $returncode != 1 && test $returncode != 250
   18 do
   19 exec 3>&1
   20 value=`$DIALOG --clear --ok-label "Create" \
   21 	  --backtitle "$backtitle" \
   22 	  --help-button \
   23 	  --help-label "Script" \
   24 	  --default-item "$defaultitem" \
   25 	  --inputmenu "Originally I designed --inputmenu for a \
   26 configuration purpose. Here is a possible piece of a configuration program." \
   27 20 60 10 \
   28 	"Username:"	"$user" \
   29 	"UID:"		"$uid"  \
   30 	"GID:"		"$gid"  \
   31 	"HOME:"		"$home" \
   32 2>&1 1>&3`
   33 returncode=$?
   34 exec 3>&-
   35 
   36 	case $returncode in
   37 	2)
   38 		"$DIALOG" \
   39 		--textbox "$0" 0 0
   40 		;;
   41 	1)
   42 		"$DIALOG" \
   43 		--clear \
   44 		--backtitle "$backtitle" \
   45 		--yesno "Really quit?" 10 30
   46 		case $? in
   47 		0)
   48 			break
   49 			;;
   50 		1)
   51 			returncode=99
   52 			;;
   53 		esac
   54 		;;
   55 	0)
   56 		"$DIALOG" \
   57 		--clear \
   58 		--backtitle "$backtitle" \
   59 		--msgbox "useradd \n\
   60 			-d $home \n\
   61 			-u $uid \n\
   62 			-g $gid \n\
   63 			$user" 10 40
   64 		;;
   65 	3)
   66 		tag=`echo "$value" |sed -e 's/^RENAMED //' -e 's/:.*/:/'`
   67 		item=`echo "$value" |sed -e 's/^.*:[ ]*//' -e 's/[ ]*$//'`
   68 
   69 		case "$tag" in
   70 		Username:)
   71 			user="$item"
   72 			;;
   73 		UID:)
   74 			uid="$item"
   75 			;;
   76 		GID:)
   77 			gid="$item"
   78 			;;
   79 		HOME:)
   80 			home="$item"
   81 			;;
   82 		*)
   83 			tag=
   84 			;;
   85 		esac
   86 		test -n "$tag" && defaultitem="$tag"
   87 		;;
   88 
   89        255)
   90                 echo "ESC pressed."
   91                 break
   92                 ;;
   93 
   94 	esac
   95 done
