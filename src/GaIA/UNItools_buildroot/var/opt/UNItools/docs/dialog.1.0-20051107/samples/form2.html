0 #! /bin/sh
    1 # $Id: form2,v 1.5 2004/09/12 22:37:31 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 backtitle="An Example for the use of --form:"
    5 
    6 ids=`id|sed -e 's/([^)]*)//g'`
    7 uid=`echo "$ids" | sed -e 's/^uid=//' -e 's/ .*//'`
    8 gid=`echo "$ids" | sed -e 's/^.* gid=//' -e 's/ .*//'`
    9 
   10 user="$USER"
   11 home="$HOME"
   12 
   13 returncode=0
   14 while test $returncode != 1 && test $returncode != 250
   15 do
   16 exec 3>&1
   17 value=`$DIALOG --ok-label "Submit" \
   18 	  --backtitle "$backtitle" \
   19 	  --form "Here is a possible piece of a configuration program." \
   20 0 0 10 \
   21 	"1 Username:"  1 1	"$user" 1  12 10 10 \
   22 	"1 UID:"       2 1	"$uid"  2  12 10 10 \
   23 	"1 GID:"       3 1	"$gid"  3  12 10 10 \
   24 	"1 HOME:"      4 1	"$home" 4  12 10 10 \
   25 	"2 Username:"  5 1	"$user" 5  12 10 10 \
   26 	"2 UID:"       6 1	"$uid"  6  12 10 10 \
   27 	"2 GID:"       7 1	"$gid"  7  12 10 10 \
   28 	"2 HOME:"      8 1	"$home" 8  12 10 10 \
   29 	"3 Username:"  9 1	"$user" 9  12 10 10 \
   30 	"3 UID:"      10 1	"$uid"  10 12 10 10 \
   31 	"3 GID:"      11 1	"$gid"  11 12 10 10 \
   32 	"3 HOME:"     12 1	"$home" 12 12 10 10 \
   33 	"4 Username:" 13 1	"$user" 13 12 10 10 \
   34 	"4 UID:"      14 1	"$uid"  14 12 10 10 \
   35 	"4 GID:"      15 1	"$gid"  15 12 10 10 \
   36 	"4 HOME:"     16 1	"$home" 16 12 10 10 \
   37 	"5 Username:" 17 1	"$user" 17 12 10 10 \
   38 	"5 UID:"      18 1	"$uid"  18 12 10 10 \
   39 	"5 GID:"      19 1	"$gid"  19 12 10 10 \
   40 	"5 HOME:"     20 1	"$home" 20 12 10 10 \
   41 	"6 Username:" 21 1	"$user" 21 12 10 10 \
   42 	"6 UID:"      22 1	"$uid"  22 12 10 10 \
   43 	"6 GID:"      23 1	"$gid"  23 12 10 10 \
   44 	"6 HOME:"     24 1	"$home" 24 12 10 10 \
   45 	"7 Username:" 25 1	"$user" 25 12 10 10 \
   46 	"7 UID:"      26 1	"$uid"  26 12 10 10 \
   47 	"7 GID:"      27 1	"$gid"  27 12 10 10 \
   48 	"7 HOME:"     28 1	"$home" 28 12 10 10 \
   49 2>&1 1>&3`
   50 returncode=$?
   51 exec 3>&-
   52 
   53 show=`echo "$value" |sed -e 's/^/	/'`
   54 
   55 	case $returncode in
   56 	1)
   57 		"$DIALOG" \
   58 		--clear \
   59 		--backtitle "$backtitle" \
   60 		--yesno "Really quit?" 10 30
   61 		case $? in
   62 		0)
   63 			break
   64 			;;
   65 		1)
   66 			returncode=99
   67 			;;
   68 		esac
   69 		;;
   70 	0)
   71 		"$DIALOG" \
   72 		--clear \
   73 		--backtitle "$backtitle" --no-collapse --cr-wrap \
   74 		--msgbox "Resulting data:\n\
   75 $show" 0 0
   76 		;;
   77 	*)
   78 		echo "Return code was $returncode"
   79 		exit
   80 		;;
   81 	esac
   82 done
