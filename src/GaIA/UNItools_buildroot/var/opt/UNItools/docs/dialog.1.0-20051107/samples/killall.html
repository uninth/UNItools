0 #! /bin/sh
    1 # $Id: killall,v 1.2 2004/09/17 19:56:18 tom Exp $
    2 # Linux has a program that does this correctly.
    3 for prog in $*
    4 do
    5 	pid=`ps -a |fgrep $prog |fgrep -v fgrep|sed -e 's/^[ ]*//' -e 's/ .*//' `
    6 	if test -n "$pid" ; then
    7 		echo killing pid=$pid, $prog
    8 		kill -1 $pid || \
    9 		kill -15 $pid || \
   10 		kill -9 $pid
   11 	fi
   12 done
