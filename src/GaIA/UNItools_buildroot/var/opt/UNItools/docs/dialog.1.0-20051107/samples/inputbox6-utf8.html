0 #!/bin/sh
    1 # $Id: inputbox6-utf8,v 1.1 2003/08/15 19:40:37 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 tempfile=`tempfile 2>/dev/null` || tempfile=/tmp/test$$
    5 trap "rm -f $tempfile" 0 1 2 5 15
    6 
    7 case none"$LANG$LC_ALL$LC_CTYPE" in
    8 *UTF-8*)
    9 	;;
   10 *)
   11 	echo "This script must be run in a UTF-8 locale"
   12 	exit 1
   13 	;;
   14 esac
   15 
   16 $DIALOG \
   17 --title    `printf "\xe3\x81\x82\xe3\x81\x84\xe3\x81\x86\xe3\x81\x88\xe3\x81\x8a"` \
   18 --inputbox `printf "\xe3\x81\x82\xe3\x81\x84\xe3\x81\x86\xe3\x81\x88\xe3\x81\x8a"` \
   19 10 20 2>$tempfile
   20 
   21 retval=$?
   22 
   23 case $retval in
   24   0)
   25     echo "Input string is \"`cat $tempfile`\"";;
   26   1)
   27     echo "Cancel pressed.";;
   28   255)
   29     if test -s $tempfile ; then
   30       cat $tempfile
   31     else
   32       echo "ESC pressed."
   33     fi
   34     ;;
   35 esac
