0 #!/bin/sh
    1 # $Id: rotated-data,v 1.1 2004/12/19 16:43:47 tom Exp $
    2 # Rotate the second parameter's data by the given shift count.
    3 if test $# != 0
    4 then
    5 	case $1 in
    6 	[1-9]*)
    7 		left=$1
    8 		next=`expr $left + 1`
    9 		shift 1
   10 		;;
   11 	*)
   12 		left=1
   13 		next=2
   14 		;;
   15 	esac
   16 	char=`echo "$@" | cut -b -${left}`
   17 	data=`echo "$@" | cut -b ${next}-`
   18 
   19 	printf "%s%s\n" $data $char
   20 else
   21 	echo
   22 fi
