   21 save() {
   22   if [ -f $1 ]
   23   then
   24     mv $1 $1~
   25   fi
   26 }

   28 # Returns $PKG_TO_REMOVE
   29 find_incompatible_packages() {
   30   dialog --backtitle "$TITLE" --infobox "Vediamo innanzitutto se c'e' qualche pacchetto di troppo ... ;-)" 0 0
   31 
   32   PKG_TO_REMOVE=""
   33   for i in sendmail smailcfg cnews
   34   do
   35     if [ -f /var/adm/packages/$i ]
   36     then
   37       PKG_TO_REMOVE="$PKG_TO_REMOVE $i"
   38     fi
   39   done
   40 }

   42 # Returns $PKG_TO_INSTALL
   43 find_necessary_packages() {
   44   dialog --backtitle "$TITLE" --infobox "Controlliamo se manca qualche pacchetto..." 0 0
   45 
   46   PKG_TO_INSTALL=""
   47   for i in tcpip mailx inn
   48   do
   49     if [ ! -f /var/adm/packages/$i ]
   50     then
   51       PKG_TO_INSTALL="$PKG_TO_INSTALL $i"
   52     fi
   53   done
   54 }

   56 # Args: user, editor
   57 configure_trn() {
   58   dialog --backtitle "$TITLE" --infobox "Configuro trn all'user $1..." 0 0
   59 
   60   SHELL_USED="`basename \`cut /etc/passwd -f1,7 -d: | grep $1: | cut -f2 -d:\``"
   61   USER_HOME="`cut /etc/passwd -f1,6 -d: | grep $1: | cut -f2 -d:`"
   62   TRNRC="$USER_HOME/.trnrc"
   63 
   64   case $SHELL_USED in
   65   bash)
   66     if [ -f $USER_HOME/.bash_profile ]
   67     then
   68       PROF=$USER_HOME/.bash_profile
   69     else
   70       PROF=$USER_HOME/.profile
   71     fi;;
   72   zsh)
   73     PROF=$USER_HOME/.zprofile;;
   74   *)
   75     PROF=$USER_HOME/.profile;;
   76   esac
   77 
   78   if [ ! -f $PROF ]
   79   then
   80     :>$PROF
   81     chown $1.mail $PROF
   82   elif [ ! "`grep -x \". ~/.trnrc\" $PROF`" ]
   83   then
   84     echo -e "\n. ~/.trnrc" >>$PROF
   85   fi
   86 
   87   save $TRNRC
   88 
   89   echo "EDITOR=\"$2\"" > $TRNRC
   90   cat copi.trnrc  >> $TRNRC
   91   chown $1.mail $TRNRC
   92 }
