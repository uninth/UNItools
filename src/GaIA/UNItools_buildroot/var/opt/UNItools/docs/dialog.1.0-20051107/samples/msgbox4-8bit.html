0 #!/bin/sh
    1 # $Id: msgbox4-8bit,v 1.1 2004/12/19 17:02:40 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 : ${DIALOG_OK=0}
    5 : ${DIALOG_CANCEL=1}
    6 : ${DIALOG_ESC=255}
    7 
    8 . ./testdata-8bit
    9 
   10 width=30
   11 while test $width != 61
   12 do
   13 $DIALOG --title "MESSAGE BOX (width $width)" --no-collapse --colors \
   14         --msgbox "\
   15 This sample uses characters which are non-printing in POSIX locale.
   16 
   17 \Z1`./rotated-data 0 $SAMPLE` \Z2`./rotated-data 1 $SAMPLE` \Z3`./rotated-data 2 $SAMPLE` \Z4`./rotated-data 3 $SAMPLE`\Zn
   18 
   19 Hi, this is a simple message box.  You can use this to \
   20 display any message you like.  The box will remain until \
   21 you press the ENTER key." 22 $width
   22 retval=$?
   23 
   24 case $retval in
   25   $DIALOG_CANCEL)
   26     echo "Cancel pressed.";exit;;
   27   $DIALOG_ESC)
   28     echo "ESC pressed.";exit;;
   29 esac
   30 
   31 width=`expr $width + 1`
   32 
   33 done
