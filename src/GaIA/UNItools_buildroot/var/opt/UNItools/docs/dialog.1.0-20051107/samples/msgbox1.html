0 #!/bin/sh
    1 # $Id: msgbox1,v 1.5 2004/06/06 13:40:58 tom Exp $
    2 : ${DIALOG=dialog}
    3 
    4 $DIALOG --title "MESSAGE BOX" --clear \
    5         --msgbox "Hi, this is a simple message box. You can use this to
    6                   display any message you like. The box will remain until
    7                   you press the ENTER key.  This box is being displayed
    8                   with dialogs default aspect ratio of 9." 0 0
    9 test $? = 255 && exit
   10 
   11 $DIALOG --aspect 12 --title "MESSAGE BOX  aspect=12" --clear \
   12         --msgbox "Hi, this is a simple message box. You can use this to
   13                   display any message you like. The box will remain until
   14                   you press the ENTER key.  This box is being displayed
   15                   with an aspect ratio of 12." 0 0
   16 test $? = 255 && exit
   17 
   18 $DIALOG --aspect 6 --title "MESSAGE BOX  aspect=6" --clear \
   19         --msgbox "Hi, this is a simple message box. You can use this to
   20                   display any message you like. The box will remain until
   21                   you press the ENTER key.  This box is being displayed
   22                   with an aspect ratio of 6." 0 0
   23 test $? = 255 && exit
   24 
   25 $DIALOG --aspect 6 --cr-wrap --title "MESSAGE BOX aspect=6 with --cr-wrap" \
   26         --clear --msgbox "\
   27            Hi, this is a simple
   28            message box. You can
   29           use this to display any
   30              message you like.
   31 The box will remain until you press the ENTER key.  This box is being displayed with an aspect ratio of 6, and using --cr-wrap.\n" 0 0
   32 test $? = 255 && exit
